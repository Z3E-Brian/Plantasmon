  <action>
    Create `src/screens/calendar/CalendarScreen.tsx` — a full-screen calendar view using `react-native-calendars`.

    **Color scheme (per D-12 — each event type gets a different dot color):**
    - Water (`water`): `#4A90D9` (blue) — D-09
    - Identify (`identify`): `#50C878` (green) — D-10
    - Achievement (`achievement`): `#FFD700` (gold) 
    - Mission (`mission`): `#9B59B6` (purple) — D-11
    
    **Component structure:**
    ```tsx
    import { useCallback, useEffect, useState } from "react";
    import { View, Text, FlatList, StyleSheet, ActivityIndicator } from "react-native";
    import { Calendar, DateData, MarkedDates } from "react-native-calendars";
    import { useSafeAreaInsets } from "react-native-safe-area-context";
    import ScreenWrapper from "@/src/components/screenWrapper/ScreenWrapper";
    import { getCurrentUserId } from "@/src/services/userService";
    import { getUserActivities, toActivityData, type ActivityEventWithId } from "@/src/services/activityService";
    import { ActivityFeed, ActivityData } from "@/src/components/profile/ActivityFeed";
    import { COLORS } from "@/src/constants/theme";
    ```

    **State:**
    - `allActivities: ActivityEventWithId[]` — raw events from Firestore
    - `selectedDate: string` — currently selected date (YYYY-MM-DD)
    - `selectedEvents: ActivityData[]` — events for selected date (filtered + converted)
    - `loading: boolean` — loading state
    - `eventDates: MarkedDates` — computed markers for the calendar

    **On mount (useEffect with getCurrentUserId):**
    1. Get userId from `getCurrentUserId()`
    2. Call `getUserActivities(userId, 200)` — get enough events for meaningful calendar
    3. Compute `eventDates` by grouping activities by date (YYYY-MM-DD from timestamp):
       ```typescript
       const marked: MarkedDates = {};
       const dateGroups: Record<string, { water: number; identify: number; achievement: number; mission: number }> = {};
       
       for (const event of allActivities) {
         const dateStr = formatDate(event.timestamp); // YYYY-MM-DD
         if (!dateGroups[dateStr]) dateGroups[dateStr] = { water: 0, identify: 0, achievement: 0, mission: 0 };
         dateGroups[dateStr][event.type]++;
       }
       
       for (const [date, counts] of Object.entries(dateGroups)) {
         const dots = [];
         if (counts.water > 0) dots.push({ key: "water", color: "#4A90D9" });
         if (counts.identify > 0) dots.push({ key: "identify", color: "#50C878" });
         if (counts.achievement > 0) dots.push({ key: "achievement", color: "#FFD700" });
         if (counts.mission > 0) dots.push({ key: "mission", color: "#9B59B6" });
         marked[date] = { dots, selected: date === selectedDate };
       }
       ```
    4. Set `eventDates` and `loading = false`

    **On day press (selectedDate change):**
    Filter `allActivities` by the selected date → map via `toActivityData()` → set `selectedEvents`.

    **Render:**
    ```
    <ScreenWrapper>
      <View style={styles.container}>
        <Text style={styles.title}>📅 Calendario de actividades</Text>
        
        <Calendar
          markedDates={eventDates}
          markingType="multi-dot"
          onDayPress={(day: DateData) => setSelectedDate(day.dateString)}
          theme={{
            todayTextColor: COLORS.primary,
            selectedDayBackgroundColor: COLORS.primary + "20",
            dotColor: COLORS.primary,
            arrowColor: COLORS.primary,
          }}
        />
        
        {/* Legend */}
        <View style={styles.legend}>
          {[
            { label: "Riego", color: "#4A90D9" },
            { label: "Identificación", color: "#50C878" },
            { label: "Logro", color: "#FFD700" },
            { label: "Misión", color: "#9B59B6" },
          ].map((item) => (
            <View key={item.label} style={styles.legendItem}>
              <View style={[styles.legendDot, { backgroundColor: item.color }]} />
              <Text style={styles.legendLabel}>{item.label}</Text>
            </View>
          ))}
        </View>
        
        {/* Selected day events */}
        <Text style={styles.sectionTitle}>
          Eventos {selectedDate ? formatDisplayDate(selectedDate) : ""}
        </Text>
        {selectedEvents.length > 0 ? (
          <ActivityFeed activities={selectedEvents} />
        ) : (
          <Text style={styles.emptyText}>No hay actividades este día</Text>
        )}
      </View>
    </ScreenWrapper>
    ```

    **Helpers:**
    - `formatDate(date: Date): string` — returns YYYY-MM-DD (zero-padded month/day)
    - `formatDisplayDate(dateStr: string): string` — converts "2026-05-23" to "23 may 2026" (Spanish locale, short month names: ene, feb, mar, abr, may, jun, jul, ago, sep, oct, nov, dic)

    **Styles:** Use StyleSheet.create with:
    - container: flex 1, padding 16
    - title: fontSize 22, fontWeight bold, marginBottom 16
    - legend: flexDirection row, justifyContent space-around, marginVertical 12, padding 8, borderRadius 12, backgroundColor rgba(0,0,0,0.03)
    - legendItem: flexDirection row, alignItems center, gap 4
    - legendDot: width 10, height 10, borderRadius 5
    - legendLabel: fontSize 12
    - sectionTitle: fontSize 16, fontWeight 600, marginVertical 12
    - emptyText: textAlign center, color gray, marginTop 20

    **Spanish locale config for react-native-calendars:**
    Add at the top of the component:

    ```typescript
      monthNamesShort: ["Ene", "Feb", "Mar", "Abr", "May", "Jun", "Jul", "Ago", "Sep", "Oct", "Nov", "Dic"],
      dayNames: ["Domingo", "Lunes", "Martes", "Miércoles", "Jueves", "Viernes", "Sábado"],
      dayNamesShort: ["Dom", "Lun", "Mar", "Mié", "Jue", "Vie", "Sáb"],
      today: "Hoy",
    };
    LocaleConfig.defaultLocale = "es";
    ```
